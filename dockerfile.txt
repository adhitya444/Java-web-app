# Step 1: Use a lightweight Tomcat 10 image
FROM tomcat:10.1-jdk17-openjdk-slim

# Step 2: Remove default apps to avoid conflicts
RUN rm -rf /usr/local/tomcat/webapps/*

# Step 3: Copy the WAR file Maven just built into the container
# Jenkins runs 'docker build' from the workspace root where 'target' exists
COPY target/Java-web-app.war /usr/local/tomcat/webapps/ROOT.war

# Step 4: Expose the port
EXPOSE 8080

# Step 5: Start Tomcat
CMD ["catalina.sh", "run"]
